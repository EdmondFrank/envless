import ReleaseCard from "../../components/ReleaseCard.astro";
import { getReleases, getLatestRelease } from "../../lib/releases.ts";

export const releases = await getReleases();
export const latest = getLatestRelease(releases);

export const pickMetric = (bench, name) =>
  bench && bench[name] ? `${bench[name].value} ${bench[name].unit}` : "—";

The list below is built at site-build time from the GitHub Releases API,
joined to `bench/history.jsonl`. New tags appear here automatically on
the next docs deploy — no manual updates per release. See
[Contributing → Release process](/envless/contributing/#release-process)
for the workflow, and [Benchmarks](/envless/benchmarks/) for the
methodology behind the perf-delta tables.

## Latest

{latest === null ? (
  <p><em>No releases have been published yet.</em></p>
) : (
  <ReleaseCard release={latest} />
)}
